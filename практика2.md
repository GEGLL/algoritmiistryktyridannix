Подробное описание структур данных: кучи и хеш-таблицы
1. Бинарные кучи
Бинарная куча — это фундаментальная структура данных, представляющая собой полное бинарное дерево, которое удовлетворяет свойству кучи. В min-куче значение каждого узла меньше или равно значениям его потомков, в max-куче — наоборот.

Преимущества:

Эффективные операции вставки и извлечения минимума/максимума (O(log n))

Простота реализации на массиве

Кэш-эффективность благодаря последовательному хранению

Недостатки:

Медленное слияние двух куч (O(n))

Ограниченная функциональность по сравнению с более сложными кучами

Реализации:
Python:

python
import heapq

# Создание min-кучи
heap = []
heapq.heappush(heap, 5)
heapq.heappush(heap, 2)
heapq.heappush(heap, 8)
heapq.heappush(heap, 1)

print("Min element:", heapq.heappop(heap))  # 1
print("Next min:", heapq.heappop(heap))     # 2

# Max-куча через отрицательные значения
max_heap = []
heapq.heappush(max_heap, -10)
heapq.heappush(max_heap, -5)
heapq.heappush(max_heap, -15)
print("Max element:", -heapq.heappop(max_heap))  # 15
C++:

cpp
#include <iostream>
#include <queue>
#include <vector>

int main() {
    // Min-куча
    std::priority_queue<int, std::vector<int>, std::greater<int>> minHeap;
    minHeap.push(5);
    minHeap.push(2);
    minHeap.push(8);
    minHeap.push(1);
    
    std::cout << "Min element: " << minHeap.top() << std::endl;  // 1
    minHeap.pop();
    std::cout << "Next min: " << minHeap.top() << std::endl;     // 2
    
    // Max-куча (по умолчанию)
    std::priority_queue<int> maxHeap;
    maxHeap.push(10);
    maxHeap.push(5);
    maxHeap.push(15);
    
    std::cout << "Max element: " << maxHeap.top() << std::endl;  // 15
    
    return 0;
}
Java:

java
import java.util.PriorityQueue;

public class BinaryHeapExample {
    public static void main(String[] args) {
        // Min-куча
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        minHeap.add(5);
        minHeap.add(2);
        minHeap.add(8);
        minHeap.add(1);
        
        System.out.println("Min element: " + minHeap.poll());  // 1
        System.out.println("Next min: " + minHeap.poll());     // 2
        
        // Max-куча через компаратор
        PriorityQueue<Integer> maxHeap = new PriorityQueue<>((a, b) -> b - a);
        maxHeap.add(10);
        maxHeap.add(5);
        maxHeap.add(15);
        
        System.out.println("Max element: " + maxHeap.poll());  // 15
    }
}
